from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple, Any
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import json
import pickle
from pathlib import Path

@dataclass
class BacktestConfig:
    """回测配置"""
    start_date: datetime
    end_date: datetime
    initial_capital: float = 100000
    commission_rate: float = 0.001
    slippage_rate: float = 0.0005
    min_commission: float = 5.0
    max_position_size: float = 0.1
    max_portfolio_risk: float = 0.02
    risk_free_rate: float = 0.03
    benchmark: str = "SH000300"
    rebalance_frequency: str = "daily"  # daily, weekly, monthly
    
@dataclass
class TradeRecord:
    """交易记录"""
    timestamp: datetime
    symbol: str
    action: str  # BUY, SELL
    quantity: int
    price: float
    commission: float
    slippage: float
    total_cost: float
    portfolio_value: float
    position_size: float
    
@dataclass
class PortfolioSnapshot:
    """组合快照"""
    timestamp: datetime
    total_value: float
    cash: float
    positions: Dict[str, int]
    weights: Dict[str, float]
    returns: float
    cumulative_returns: float
    drawdown: float
    volatility: float

class BacktestEngine:
    """回测引擎基类"""
    
    def __init__(self, config: BacktestConfig):
        self.config = config
        self.portfolio = self.initialize_portfolio()
        self.trade_history = []
        self.portfolio_history = []
        self.signals_history = []
        self.performance_metrics = {}
        self.is_running = False
        
    def initialize_portfolio(self) -> Dict[str, Any]:
        """初始化组合"""
        return {
            'cash': self.config.initial_capital,
            'positions': {},
            'total_value': self.config.initial_capital,
            'weights': {}
        }
    
    @abstractmethod
    def generate_signals(self, data: pd.DataFrame, timestamp: datetime) -> List[Dict[str, Any]]:
        """生成交易信号"""
        pass
    
    @abstractmethod
    def calculate_position_size(self, signal: Dict[str, Any], current_price: float) -> int:
        """计算仓位大小"""
        pass
    
    def run_backtest(self, market_data: Dict[str, pd.DataFrame], 
                    strategy_params: Dict[str, Any] = None) -> Dict[str, Any]:
        """运行回测"""
        print(f"开始回测: {self.config.start_date} 到 {self.config.end_date}")
        self.is_running = True
        
        # 获取交易日历
        trading_dates = self.get_trading_dates(market_data)
        
        for current_date in trading_dates:
            if not self.is_running:
                break
                
            # 获取当日数据
            daily_data = self.get_daily_data(market_data, current_date)
            
            if daily_data.empty:
                continue
            
            # 生成交易信号
            signals = self.generate_signals(daily_data, current_date)
            self.signals_history.append({
                'timestamp': current_date,
                'signals': signals
            })
            
            # 执行交易
            for signal in signals:
                self.execute_trade(signal, daily_data, current_date)
            
            # 更新组合价值
            self.update_portfolio_value(daily_data, current_date)
            
            # 记录组合状态
            self.record_portfolio_snapshot(current_date)
        
        # 计算性能指标
        self.calculate_performance_metrics()
        
        return self.generate_backtest_report()
    
    def get_trading_dates(self, market_data: Dict[str, pd.DataFrame]) -> List[datetime]:
        """获取交易日历"""
        # 使用第一个股票的数据日期作为交易日历
        symbol = list(market_data.keys())[0]
        data = market_data[symbol]
        
        # 过滤日期范围
        mask = (data.index >= self.config.start_date) & (data.index <= self.config.end_date)
        trading_dates = data.index[mask].tolist()
        
        return trading_dates
    
    def get_daily_data(self, market_data: Dict[str, pd.DataFrame], 
                      date: datetime) -> pd.DataFrame:
        """获取当日市场数据"""
        daily_data = {}
        
        for symbol, data in market_data.items():
            if date in data.index:
                daily_data[symbol] = data.loc[date]
        
        return pd.DataFrame(daily_data).T
    
    def execute_trade(self, signal: Dict[str, Any], market_data: pd.DataFrame, 
                     timestamp: datetime):
        """执行交易"""
        symbol = signal['symbol']
        action = signal['action']
        quantity = signal.get('quantity', 0)
        
        if symbol not in market_data.index:
            return
        
        # 获取当前价格
        current_data = market_data.loc[symbol]
        current_price = current_data['close']
        
        # 计算交易成本和滑点
        commission = max(self.config.min_commission, 
                        quantity * current_price * self.config.commission_rate)
        slippage = quantity * current_price * self.config.slippage_rate
        total_cost = quantity * current_price + commission + slippage
        
        # 执行买入
        if action == 'BUY':
            if self.portfolio['cash'] >= total_cost:
                # 更新现金
                self.portfolio['cash'] -= total_cost
                
                # 更新持仓
                if symbol in self.portfolio['positions']:
                    self.portfolio['positions'][symbol] += quantity
                else:
                    self.portfolio['positions'][symbol] = quantity
                
                # 记录交易
                trade = TradeRecord(
                    timestamp=timestamp,
                    symbol=symbol,
                    action='BUY',
                    quantity=quantity,
                    price=current_price,
                    commission=commission,
                    slippage=slippage,
                    total_cost=total_cost,
                    portfolio_value=self.portfolio['total_value'],
                    position_size=quantity * current_price / self.portfolio['total_value']
                )
                self.trade_history.append(trade)
                
                print(f"买入: {symbol} {quantity}股 @ {current_price:.2f}, 成本: {total_cost:.2f}")
        
        # 执行卖出
        elif action == 'SELL':
            if symbol in self.portfolio['positions'] and self.portfolio['positions'][symbol] >= quantity:
                # 更新现金
                self.portfolio['cash'] += (quantity * current_price - commission - slippage)
                
                # 更新持仓
                self.portfolio['positions'][symbol] -= quantity
                if self.portfolio['positions'][symbol] == 0:
                    del self.portfolio['positions'][symbol]
                
                # 记录交易
                trade = TradeRecord(
                    timestamp=timestamp,
                    symbol=symbol,
                    action='SELL',
                    quantity=quantity,
                    price=current_price,
                    commission=commission,
                    slippage=slippage,
                    total_cost=quantity * current_price - commission - slippage,
                    portfolio_value=self.portfolio['total_value'],
                    position_size=quantity * current_price / self.portfolio['total_value']
                )
                self.trade_history.append(trade)
                
                print(f"卖出: {symbol} {quantity}股 @ {current_price:.2f}, 收入: {total_cost:.2f}")
    
    def update_portfolio_value(self, market_data: pd.DataFrame, timestamp: datetime):
        """更新组合价值"""
        total_value = self.portfolio['cash']
        
        for symbol, position in self.portfolio['positions'].items():
            if symbol in market_data.index:
                current_price = market_data.loc[symbol, 'close']
                total_value += position * current_price
        
        self.portfolio['total_value'] = total_value
        
        # 更新权重
        weights = {}
        for symbol, position in self.portfolio['positions'].items():
            if symbol in market_data.index:
                current_price = market_data.loc[symbol, 'close']
                position_value = position * current_price
                weights[symbol] = position_value / total_value if total_value > 0 else 0
        
        self.portfolio['weights'] = weights
    
    def record_portfolio_snapshot(self, timestamp: datetime):
        """记录组合快照"""
        snapshot = PortfolioSnapshot(
            timestamp=timestamp,
            total_value=self.portfolio['total_value'],
            cash=self.portfolio['cash'],
            positions=self.portfolio['positions'].copy(),
            weights=self.portfolio['weights'].copy(),
            returns=0.0,
            cumulative_returns=0.0,
            drawdown=0.0,
            volatility=0.0
        )
        
        self.portfolio_history.append(snapshot)
    
    def calculate_performance_metrics(self):
        """计算性能指标"""
        if not self.portfolio_history:
            return
        
        # 转换为DataFrame
        portfolio_df = pd.DataFrame([
            {
                'timestamp': s.timestamp,
                'total_value': s.total_value,
                'cash': s.cash
            } for s in self.portfolio_history
        ])
        
        # 计算收益率
        portfolio_df['returns'] = portfolio_df['total_value'].pct_change()
        portfolio_df['cumulative_returns'] = (1 + portfolio_df['returns'].fillna(0)).cumprod() - 1
        
        # 更新快照
        for i, snapshot in enumerate(self.portfolio_history):
            if i > 0:
                snapshot.returns = portfolio_df.iloc[i]['returns']
                snapshot.cumulative_returns = portfolio_df.iloc[i]['cumulative_returns']
        
        # 计算详细指标
        self.performance_metrics = self.calculate_detailed_metrics(portfolio_df['returns'])
    
    def calculate_detailed_metrics(self, returns: pd.Series) -> Dict[str, float]:
        """计算详细性能指标"""
        metrics = {}
        
        # 基本指标
        total_return = (self.portfolio_history[-1].total_value - self.config.initial_capital) / self.config.initial_capital
        metrics['total_return'] = total_return
        
        # 年化收益率
        days = len(returns)
        metrics['annual_return'] = (1 + total_return) ** (252 / days) - 1 if days > 0 else 0
        
        # 年化波动率
        metrics['annual_volatility'] = returns.std() * np.sqrt(252)
        
        # 夏普比率
        excess_returns = metrics['annual_return'] - self.config.risk_free_rate
        metrics['sharpe_ratio'] = excess_returns / metrics['annual_volatility'] if metrics['annual_volatility'] > 0 else 0
        
        # 最大回撤
        cumulative = (1 + returns.fillna(0)).cumprod()
        running_max = cumulative.expanding().max()
        drawdown = (cumulative - running_max) / running_max
        metrics['max_drawdown'] = drawdown.min()
        
        # 卡尔马比率
        if abs(metrics['max_drawdown']) > 0:
            metrics['calmar_ratio'] = metrics['annual_return'] / abs(metrics['max_drawdown'])
        else:
            metrics['calmar_ratio'] = 0
        
        # 胜率
        if len(self.trade_history) > 0:
            winning_trades = [t for t in self.trade_history if t.action == 'SELL']
            metrics['win_rate'] = len(winning_trades) / len(self.trade_history)
        else:
            metrics['win_rate'] = 0
        
        # 交易统计
        metrics['total_trades'] = len(self.trade_history)
        metrics['avg_trade_return'] = np.mean([t.total_cost for t in self.trade_history]) if self.trade_history else 0
        
        return metrics
    
    def generate_backtest_report(self) -> Dict[str, Any]:
        """生成回测报告"""
        return {
            'config': self.config,
            'performance_metrics': self.performance_metrics,
            'trade_history': self.trade_history,
            'portfolio_history': self.portfolio_history,
            'signals_history': self.signals_history,
            'final_portfolio_value': self.portfolio['total_value'],
            'total_return': self.performance_metrics.get('total_return', 0),
            'sharpe_ratio': self.performance_metrics.get('sharpe_ratio', 0),
            'max_drawdown': self.performance_metrics.get('max_drawdown', 0)
        }

class AdvancedBacktestEngine(BacktestEngine):
    """高级回测引擎"""
    
    def __init__(self, config: BacktestConfig):
        super().__init__(config)
        self.benchmark_data = None
        self.risk_model = None
        self.transaction_cost_model = None
        self.slippage_model = None
        self.market_impact_model = None
    
    def set_benchmark(self, benchmark_data: pd.DataFrame):
        """设置基准数据"""
        self.benchmark_data = benchmark_data
    
    def set_risk_model(self, risk_model):
        """设置风险模型"""
        self.risk_model = risk_model
    
    def set_transaction_cost_model(self, cost_model):
        """设置交易成本模型"""
        self.transaction_cost_model = cost_model
    
    def calculate_transaction_cost(self, symbol: str, quantity: int, 
                                 price: float, action: str) -> Dict[str, float]:
        """计算交易成本"""
        base_cost = quantity * price
        
        # 基本佣金
        commission = max(self.config.min_commission, base_cost * self.config.commission_rate)
        
        # 使用高级成本模型（如果设置）
        if self.transaction_cost_model:
            additional_costs = self.transaction_cost_model.calculate_cost(
                symbol, quantity, price, action
            )
            commission += additional_costs.get('additional_commission', 0)
        
        # 滑点
        slippage = base_cost * self.config.slippage_rate
        
        # 市场冲击（大额交易）
        if quantity * price > 100000:  # 超过10万元
            market_impact = base_cost * 0.0005  # 额外0.05%冲击成本
        else:
            market_impact = 0
        
        return {
            'commission': commission,
            'slippage': slippage,
            'market_impact': market_impact,
            'total_cost': commission + slippage + market_impact
        }
    
    def execute_trade_with_impact(self, signal: Dict[str, Any], 
                                market_data: pd.DataFrame, timestamp: datetime):
        """执行交易（考虑市场冲击）"""
        symbol = signal['symbol']
        action = signal['action']
        quantity = signal.get('quantity', 0)
        
        if symbol not in market_data.index:
            return
        
        # 获取当前价格
        current_data = market_data.loc[symbol]
        current_price = current_data['close']
        
        # 计算交易成本
        costs = self.calculate_transaction_cost(symbol, quantity, current_price, action)
        
        # 大额交易的价格冲击
        if quantity * current_price > 100000:
            # 模拟市场冲击导致的价格滑点
            if action == 'BUY':
                executed_price = current_price * 1.001  # 买入价格上升0.1%
            else:
                executed_price = current_price * 0.999  # 卖出价格下降0.1%
        else:
            executed_price = current_price
        
        total_transaction_cost = quantity * executed_price + costs['total_cost']
        
        # 执行交易逻辑（与基础引擎类似，但使用实际执行价格）
        if action == 'BUY' and self.portfolio['cash'] >= total_transaction_cost:
            self.portfolio['cash'] -= total_transaction_cost
            self.portfolio['positions'][symbol] = self.portfolio['positions'].get(symbol, 0) + quantity
            
        elif action == 'SELL' and symbol in self.portfolio['positions']:
            if self.portfolio['positions'][symbol] >= quantity:
                self.portfolio['cash'] += (quantity * executed_price - costs['total_cost'])
                self.portfolio['positions'][symbol] -= quantity
                if self.portfolio['positions'][symbol] == 0:
                    del self.portfolio['positions'][symbol]
        
        # 记录详细交易信息
        trade = TradeRecord(
            timestamp=timestamp,
            symbol=symbol,
            action=action,
            quantity=quantity,
            price=executed_price,
            commission=costs['commission'],
            slippage=costs['slippage'],
            total_cost=total_transaction_cost,
            portfolio_value=self.portfolio['total_value'],
            position_size=quantity * executed_price / self.portfolio['total_value']
        )
        self.trade_history.append(trade)
    
    def calculate_risk_adjusted_metrics(self, returns: pd.Series) -> Dict[str, float]:
        """计算风险调整指标"""
        metrics = {}
        
        # 基础指标
        metrics.update(self.calculate_detailed_metrics(returns))
        
        # 索提诺比率（下行风险）
        downside_returns = returns[returns < 0]
        if len(downside_returns) > 0:
            downside_std = downside_returns.std() * np.sqrt(252)
            excess_return = metrics['annual_return'] - self.config.risk_free_rate
            metrics['sortino_ratio'] = excess_return / downside_std if downside_std > 0 else 0
        else:
            metrics['sortino_ratio'] = np.inf
        
        # 信息比率（相对基准）
        if self.benchmark_data is not None:
            benchmark_returns = self.calculate_benchmark_returns()
            active_returns = returns - benchmark_returns
            tracking_error = active_returns.std() * np.sqrt(252)
            
            if tracking_error > 0:
                metrics['information_ratio'] = active_returns.mean() * 252 / tracking_error
            else:
                metrics['information_ratio'] = 0
        
        # 收益一致性
        monthly_returns = returns.resample('M').apply(lambda x: (1 + x).prod() - 1)
        metrics['consistency'] = (monthly_returns > 0).mean()
        
        # 风险价值（VaR）
        metrics['var_95'] = np.percentile(returns, 5)
        metrics['var_99'] = np.percentile(returns, 1)
        
        # 预期亏损（CVaR）
        metrics['cvar_95'] = returns[returns <= metrics['var_95']].mean()
        
        return metrics
    
    def calculate_benchmark_returns(self) -> pd.Series:
        """计算基准收益率"""
        if self.benchmark_data is None:
            return pd.Series(dtype=float)
        
        # 过滤回测期间的数据
        mask = (self.benchmark_data.index >= self.config.start_date) & \
               (self.benchmark_data.index <= self.config.end_date)
        benchmark_period = self.benchmark_data[mask]
        
        # 计算日收益率
        benchmark_returns = benchmark_period['close'].pct_change()
        
        return benchmark_returns
    
    def generate_comprehensive_report(self) -> Dict[str, Any]:
        """生成综合回测报告"""
        basic_report = self.generate_backtest_report()
        
        # 添加风险调整指标
        if self.portfolio_history:
            returns = pd.Series([s.returns for s in self.portfolio_history[1:]])
            risk_metrics = self.calculate_risk_adjusted_metrics(returns)
            basic_report['risk_metrics'] = risk_metrics
        
        # 添加基准比较
        if self.benchmark_data is not None:
            basic_report['benchmark_comparison'] = self.compare_with_benchmark()
        
        # 添加交易分析
        basic_report['trade_analysis'] = self.analyze_trades()
        
        return basic_report
    
    def compare_with_benchmark(self) -> Dict[str, Any]:
        """与基准比较"""
        if self.benchmark_data is None:
            return {}
        
        # 计算策略收益率
        strategy_returns = pd.Series([s.returns for s in self.portfolio_history[1:]])
        
        # 计算基准收益率
        benchmark_returns = self.calculate_benchmark_returns()
        
        # 对齐时间序列
        aligned_data = pd.concat([strategy_returns, benchmark_returns], axis=1).dropna()
        strategy_aligned = aligned_data.iloc[:, 0]
        benchmark_aligned = aligned_data.iloc[:, 1]
        
        # 计算相对表现
        comparison = {
            'strategy_total_return': (1 + strategy_aligned).prod() - 1,
            'benchmark_total_return': (1 + benchmark_aligned).prod() - 1,
            'excess_return': (1 + strategy_aligned).prod() - (1 + benchmark_aligned).prod(),
            'correlation': strategy_aligned.corr(benchmark_aligned),
            'beta': self.calculate_beta(strategy_aligned, benchmark_aligned),
            'alpha': self.calculate_alpha(strategy_aligned, benchmark_aligned)
        }
        
        return comparison
    
    def calculate_beta(self, strategy_returns: pd.Series, benchmark_returns: pd.Series) -> float:
        """计算Beta值"""
        covariance = np.cov(strategy_returns, benchmark_returns)[0, 1]
        benchmark_variance = np.var(benchmark_returns)
        
        return covariance / benchmark_variance if benchmark_variance > 0 else 0
    
    def calculate_alpha(self, strategy_returns: pd.Series, benchmark_returns: pd.Series) -> float:
        """计算Alpha值"""
        beta = self.calculate_beta(strategy_returns, benchmark_returns)
        strategy_return = strategy_returns.mean() * 252
        benchmark_return = benchmark_returns.mean() * 252
        
        return strategy_return - (self.config.risk_free_rate + beta * (benchmark_return - self.config.risk_free_rate))
    
    def analyze_trades(self) -> Dict[str, Any]:
        """分析交易记录"""
        if not self.trade_history:
            return {}
        
        trades_df = pd.DataFrame([
            {
                'timestamp': t.timestamp,
                'symbol': t.symbol,
                'action': t.action,
                'quantity': t.quantity,
                'price': t.price,
                'total_cost': t.total_cost,
                'commission': t.commission,
                'position_size': t.position_size
            } for t in self.trade_history
        ])
        
        analysis = {
            'total_trades': len(trades_df),
            'buy_trades': len(trades_df[trades_df['action'] == 'BUY']),
            'sell_trades': len(trades_df[trades_df['action'] == 'SELL']),
            'total_commission': trades_df['commission'].sum(),
            'avg_position_size': trades_df['position_size'].mean(),
            'max_position_size': trades_df['position_size'].max(),
            'trading_frequency': len(trades_df) / len(self.portfolio_history) if self.portfolio_history else 0
        }
        
        # 按股票分析
        symbol_analysis = trades_df.groupby('symbol').agg({
            'total_cost': ['count', 'sum', 'mean'],
            'commission': 'sum',
            'position_size': 'mean'
        }).round(4)
        
        analysis['symbol_analysis'] = symbol_analysis.to_dict()
        
        return analysis

class ParameterOptimizer:
    """参数优化器"""
    
    def __init__(self, backtest_engine: BacktestEngine):
        self.engine = backtest_engine
        self.optimization_results = []
        self.best_params = None
        self.best_performance = None
    
    def grid_search(self, param_grid: Dict[str, List[Any]], 
                   market_data: Dict[str, pd.DataFrame],
                   optimization_metric: str = 'sharpe_ratio') -> Dict[str, Any]:
        """网格搜索优化"""
        
        print("开始网格搜索参数优化...")
        
        # 生成所有参数组合
        param_combinations = self.generate_param_combinations(param_grid)
        total_combinations = len(param_combinations)
        
        print(f"总共有 {total_combinations} 种参数组合需要测试")
        
        best_metric = -np.inf
        best_params = None
        optimization_results = []
        
        for i, params in enumerate(param_combinations):
            print(f"测试参数组合 {i+1}/{total_combinations}: {params}")
            
            # 运行回测
            result = self.run_backtest_with_params(params, market_data)
            
            if result and optimization_metric in result['performance_metrics']:
                metric_value = result['performance_metrics'][optimization_metric]
                
                # 记录结果
                optimization_results.append({
                    'params': params,
                    'metrics': result['performance_metrics'],
                    'metric_value': metric_value
                })
                
                # 更新最佳参数
                if metric_value > best_metric:
                    best_metric = metric_value
                    best_params = params
                    self.best_performance = result['performance_metrics']
                    
                    print(f"  找到更好的参数组合！{optimization_metric}: {metric_value:.4f}")
        
        self.optimization_results = optimization_results
        self.best_params = best_params
        
        return {
            'best_params': best_params,
            'best_metric': best_metric,
            'optimization_results': optimization_results,
            'total_combinations': total_combinations
        }
    
    def generate_param_combinations(self, param_grid: Dict[str, List[Any]]) -> List[Dict[str, Any]]:
        """生成参数组合"""
        import itertools
        
        param_names = list(param_grid.keys())
        param_values = list(param_grid.values())
        
        combinations = list(itertools.product(*param_values))
        
        return [dict(zip(param_names, combo)) for combo in combinations]
    
    def run_backtest_with_params(self, params: Dict[str, Any], 
                               market_data: Dict[str, pd.DataFrame]) -> Dict[str, Any]:
        """使用指定参数运行回测"""
        try:
            # 创建新的策略实例（这里简化处理，实际应该根据参数创建不同策略）
            strategy_params = params
            
            # 这里假设引擎有方法根据参数调整策略
            # 实际实现中需要修改引擎使其支持参数化策略
            result = self.engine.run_backtest(market_data, strategy_params)
            
            return result
            
        except Exception as e:
            print(f"回测失败: {e}")
            return None
    
    def random_search(self, param_distributions: Dict[str, Tuple[Any, Any]], 
                     n_iterations: int, market_data: Dict[str, pd.DataFrame],
                     optimization_metric: str = 'sharpe_ratio') -> Dict[str, Any]:

        best_params = None
        optimization_results = []
        
        for i in range(n_iterations):
            # 随机生成参数

            if result and optimization_metric in result['performance_metrics']:
                metric_value = result['performance_metrics'][optimization_metric]
                
                # 记录结果
                optimization_results.append({
                    'params': params,
                    'metrics': result['performance_metrics'],

                })
                
                # 更新最佳参数
                if metric_value > best_metric:
                    best_metric = metric_value
                    best_params = params
                    self.best_performance = result['performance_metrics']
                    
                    print(f"  找到更好的参数组合！{optimization_metric}: {metric_value:.4f}")
        
        self.optimization_results = optimization_results
        self.best_params = best_params
        
        return {
            'best_params': best_params,
            'best_metric': best_metric,
            'optimization_results': optimization_results,
            'total_iterations': n_iterations
        }
    
    def generate_random_params(self, param_distributions: Dict[str, Tuple[Any, Any]]) -> Dict[str, Any]:
        """生成随机参数"""
        params = {}
        
        for param_name, (param_type, param_range) in param_distributions.items():
            if param_type == 'int':
                params[param_name] = np.random.randint(param_range[0], param_range[1] + 1)
            elif param_type == 'float':
                params[param_name] = np.random.uniform(param_range[0], param_range[1])
            elif param_type == 'choice':
                params[param_name] = np.random.choice(param_range)
        
        return params
    
    def bayesian_optimization(self, param_bounds: Dict[str, Tuple[float, float]], 
                            n_iterations: int, market_data: Dict[str, pd.DataFrame],
                            optimization_metric: str = 'sharpe_ratio') -> Dict[str, Any]:
        """贝叶斯优化"""
        
        try:
            from skopt import gp_minimize
            from skopt.space import Real, Integer
            
            print(f"开始贝叶斯参数优化（{n_iterations}次迭代）...")
            
            # 定义参数空间
            search_space = []
            param_names = []
            
            for param_name, (param_type, bounds) in param_bounds.items():
                param_names.append(param_name)
                
                if param_type == 'int':
                    search_space.append(Integer(bounds[0], bounds[1], name=param_name))
                elif param_type == 'float':
                    search_space.append(Real(bounds[0], bounds[1], name=param_name))
            
            # 定义目标函数
            def objective_function(params):
                param_dict = dict(zip(param_names, params))
                
                # 运行回测
                result = self.run_backtest_with_params(param_dict, market_data)
                
                if result and optimization_metric in result['performance_metrics']:
                    # 贝叶斯优化是求最小值，所以返回负值
                    return -result['performance_metrics'][optimization_metric]
                else:
                    return 0  # 返回一个较大的值表示失败
            
            # 运行贝叶斯优化
            result = gp_minimize(
                func=objective_function,
                dimensions=search_space,
                n_calls=n_iterations,
                n_initial_points=10,
                random_state=42
            )
            
            # 提取最佳参数
            best_params = dict(zip(param_names, result.x))
            best_metric = -result.fun  # 转换回正值
            
            self.best_params = best_params
            
            return {
                'best_params': best_params,
                'best_metric': best_metric,
                'optimization_result': result,
                'total_iterations': n_iterations
            }
            
        except ImportError:
            print("scikit-optimize未安装，请使用pip install scikit-optimize安装")
            return None
        except Exception as e:
            print(f"贝叶斯优化失败: {e}")
            return None

class GeneticAlgorithmOptimizer:
    """遗传算法优化器"""
    
    def __init__(self, population_size: int = 50, generations: int = 20,
                 mutation_rate: float = 0.1, crossover_rate: float = 0.8):
        self.population_size = population_size
        self.generations = generations
        self.mutation_rate = mutation_rate
        self.crossover_rate = crossover_rate
        self.population = []
        self.fitness_history = []
    
    def optimize(self, param_bounds: Dict[str, Any], 
                backtest_func, market_data: Dict[str, pd.DataFrame],
                optimization_metric: str = 'sharpe_ratio') -> Dict[str, Any]:
        """遗传算法优化"""
        
        print(f"开始遗传算法优化（种群大小: {self.population_size}, 代数: {self.generations}）...")
        
        # 初始化种群
        self.population = self.initialize_population(param_bounds)
        
        best_individual = None
        best_fitness = -np.inf
        
        for generation in range(self.generations):
            print(f"第 {generation + 1} 代优化...")
            
            # 评估适应度
            fitness_scores = []
            for individual in self.population:
                fitness = self.evaluate_fitness(individual, backtest_func, market_data, optimization_metric)
                fitness_scores.append(fitness)
                
                if fitness > best_fitness:
                    best_fitness = fitness
                    best_individual = individual.copy()
            
            self.fitness_history.append({
                'generation': generation,
                'best_fitness': best_fitness,
                'avg_fitness': np.mean(fitness_scores)
            })
            
            print(f"  最佳适应度: {best_fitness:.4f}")
            
            # 选择
            selected_population = self.selection(self.population, fitness_scores)
            
            # 交叉
            offspring = self.crossover(selected_population)
            
            # 变异
            mutated_offspring = self.mutation(offspring, param_bounds)
            
            # 更新种群
            self.population = mutated_offspring
        
        return {
            'best_params': best_individual,
            'best_fitness': best_fitness,
            'fitness_history': self.fitness_history
        }
    
    def initialize_population(self, param_bounds: Dict[str, Any]) -> List[Dict[str, Any]]:
        """初始化种群"""
        population = []
        
        for _ in range(self.population_size):
            individual = {}
            
            for param_name, param_config in param_bounds.items():
                if param_config['type'] == 'int':
                    individual[param_name] = np.random.randint(
                        param_config['bounds'][0], param_config['bounds'][1] + 1
                    )
                elif param_config['type'] == 'float':
                    individual[param_name] = np.random.uniform(
                        param_config['bounds'][0], param_config['bounds'][1]
                    )
                elif param_config['type'] == 'choice':
                    individual[param_name] = np.random.choice(param_config['choices'])
            
            population.append(individual)
        
        return population
    
    def evaluate_fitness(self, individual: Dict[str, Any], backtest_func,
                          market_data: Dict[str, pd.DataFrame], 
                          optimization_metric: str) -> float:
        """评估适应度"""
        try:
            # 运行回测
            result = backtest_func(individual, market_data)
            
            if result and optimization_metric in result.get('performance_metrics', {}):
                return result['performance_metrics'][optimization_metric]
            else:
                return -np.inf  # 返回负无穷表示失败
                
        except Exception as e:
            print(f"适应度评估失败: {e}")
            return -np.inf
    
    def selection(self, population: List[Dict[str, Any]], 
                   fitness_scores: List[float]) -> List[Dict[str, Any]]:
        """选择操作（锦标赛选择）"""
        selected = []
        
        for _ in range(len(population)):
            # 随机选择3个个体进行锦标赛
            tournament_indices = np.random.choice(len(population), 3, replace=False)
            tournament_fitness = [fitness_scores[i] for i in tournament_indices]
            winner_idx = tournament_indices[np.argmax(tournament_fitness)]
            selected.append(population[winner_idx].copy())
        
        return selected
    
    def crossover(self, population: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """交叉操作"""
        offspring = []
        
        for i in range(0, len(population), 2):
            parent1 = population[i]
            parent2 = population[i + 1] if i + 1 < len(population) else population[0]
            
            if np.random.random() < self.crossover_rate:
                # 单点交叉
                child1, child2 = self.single_point_crossover(parent1, parent2)
                offspring.extend([child1, child2])
            else:
                offspring.extend([parent1.copy(), parent2.copy()])
        
        return offspring[:len(population)]  # 保持种群大小
    
    def single_point_crossover(self, parent1: Dict[str, Any], 
                               parent2: Dict[str, Any]) -> Tuple[Dict[str, Any], Dict[str, Any]]:
        """单点交叉"""
        child1 = parent1.copy()
        child2 = parent2.copy()
        
        # 随机选择交叉点
        param_names = list(parent1.keys())
        crossover_point = np.random.randint(1, len(param_names))
        
        # 交换交叉点后的基因
        for i in range(crossover_point, len(param_names)):
            param_name = param_names[i]
            child1[param_name] = parent2[param_name]
            child2[param_name] = parent1[param_name]
        
        return child1, child2
    
    def mutation(self, population: List[Dict[str, Any]], 
                param_bounds: Dict[str, Any]) -> List[Dict[str, Any]]:
        """变异操作"""
        mutated_population = []
        
        for individual in population:
            mutated_individual = individual.copy()
            
            for param_name, param_config in param_bounds.items():
                if np.random.random() < self.mutation_rate:
                    # 执行变异
                    if param_config['type'] == 'int':
                        mutated_individual[param_name] = np.random.randint(
                            param_config['bounds'][0], param_config['bounds'][1] + 1
                        )
                    elif param_config['type'] == 'float':
                        mutated_individual[param_name] = np.random.uniform(
                            param_config['bounds'][0], param_config['bounds'][1]
                        )
                    elif param_config['type'] == 'choice':
                        mutated_individual[param_name] = np.random.choice(param_config['choices'])
            
            mutated_population.append(mutated_individual)
        
        return mutated_population

class OverfittingDetector:
    """过拟合检测器"""
    
    def __init__(self):
        self.detection_results = {}
        self.validation_metrics = {}
    
    def walk_forward_analysis(self, strategy_class, param_grid: Dict[str, List[Any]],
                            market_data: Dict[str, pd.DataFrame],
                            train_period: int = 252, test_period: int = 63,
                            step_size: int = 21) -> Dict[str, Any]:
        """前向分析（Walk-Forward Analysis）"""
        
        print("开始前向分析...")
        
        # 获取所有交易日期
        all_dates = self.get_trading_dates(market_data)
        total_days = len(all_dates)
        
        results = []
        
        # 滑动窗口进行训练和测试
        for start_idx in range(0, total_days - train_period - test_period, step_size):
            # 训练期
            train_start = all_dates[start_idx]
            train_end = all_dates[start_idx + train_period - 1]
            
            # 测试期
            test_start = all_dates[start_idx + train_period]
            test_end = all_dates[min(start_idx + train_period + test_period - 1, total_days - 1)]
            
            print(f"窗口 {len(results) + 1}: 训练期 {train_start.date()} - {train_end.date()}, "
                  f"测试期 {test_start.date()} - {test_end.date()}")
            
            # 训练期数据
            train_data = self.get_period_data(market_data, train_start, train_end)
            
            # 测试期数据
            test_data = self.get_period_data(market_data, test_start, test_end)
            
            # 在训练期上优化参数
            optimizer = ParameterOptimizer(strategy_class(BacktestConfig()))
            optimization_result = optimizer.grid_search(param_grid, train_data)
            
            if optimization_result['best_params']:
                # 在测试期上验证
                test_result = self.run_backtest_with_params(
                    optimization_result['best_params'], test_data
                )
                
                if test_result:
                    results.append({
                        'train_period': (train_start, train_end),
                        'test_period': (test_start, test_end),
                        'best_params': optimization_result['best_params'],
                        'train_metrics': optimization_result.get('best_performance', {}),
                        'test_metrics': test_result.get('performance_metrics', {}),
                        'performance_degradation': self.calculate_performance_degradation(
                            optimization_result.get('best_performance', {}),
                            test_result.get('performance_metrics', {})
                        )
                    })
        
        # 分析结果
        analysis = self.analyze_walk_forward_results(results)
        
        return {
            'walk_forward_results': results,
            'analysis': analysis,
            'is_overfitted': analysis.get('is_overfitted', False)
        }
    
    def monte_carlo_permutation_test(self, returns: pd.Series, 
                                   n_permutations: int = 1000) -> Dict[str, Any]:
        """蒙特卡洛置换检验"""
        
        print(f"开始蒙特卡洛置换检验（{n_permutations}次置换）...")
        
        # 计算原始夏普比率
        original_sharpe = self.calculate_sharpe_ratio(returns)
        
        # 生成随机置换
        permuted_sharpes = []
        
        for i in range(n_permutations):
            # 随机打乱收益率序列
            permuted_returns = np.random.permutation(returns.values)
            permuted_series = pd.Series(permuted_returns, index=returns.index)
            
            # 计算置换后的夏普比率
            permuted_sharpe = self.calculate_sharpe_ratio(permuted_series)
            permuted_sharpes.append(permuted_sharpe)
            
            if i % 100 == 0:
                print(f"  完成 {i}/{n_permutations} 次置换")
        
        permuted_sharpes = np.array(permuted_sharpes)
        
        # 计算p值
        p_value = (permuted_sharpes >= original_sharpe).mean()
        
        return {
            'original_sharpe': original_sharpe,
            'mean_permuted_sharpe': np.mean(permuted_sharpes),
            'std_permuted_sharpe': np.std(permuted_sharpes),
            'p_value': p_value,
            'is_significant': p_value < 0.05,
            'n_permutations': n_permutations
        }
    
    def data_mining_bias_test(self, strategy_returns: pd.Series,

        strategy_sharpe = self.calculate_sharpe_ratio(strategy_returns)
        
        # 计算随机策略的夏普比率
        random_sharpes = []
        for random_returns in random_strategy_returns:
            random_sharpe = self.calculate_sharpe_ratio(random_returns)
            random_sharpes.append(random_sharpe)
        
        random_sharpes = np.array(random_sharpes)
        
        # 计算排名
        all_sharpes = np.append(random_sharpes, strategy_sharpe)
        rank = np.sum(all_sharpes >= strategy_sharpe)
        percentile_rank = rank / len(all_sharpes)
        
        return {
            'strategy_sharpe': strategy_sharpe,
            'mean_random_sharpe': np.mean(random_sharpes),
            'std_random_sharpe': np.std(random_sharpes),

        """计算夏普比率"""
        if returns.std() == 0:

        excess_return = returns.mean() * 252 - risk_free_rate
        return excess_return / (returns.std() * np.sqrt(252))

            if train_value != 0:
                degradation[metric] = (test_value - train_value) / abs(train_value)
            else:
                degradation[metric] = 0
        
        return degradation

class PerformanceEvaluator:
    """性能评估器"""
    
    def __init__(self, risk_free_rate: float = 0.03):
        self.risk_free_rate = risk_free_rate
        self.evaluation_results = {}
    
    def comprehensive_evaluation(self, returns: pd.Series, 
                               benchmark_returns: pd.Series = None) -> Dict[str, Any]:
        """综合性能评估"""
        
        evaluation = {}
        
        # 基本统计指标
        evaluation['basic_stats'] = self.calculate_basic_stats(returns)
        
        # 风险调整指标
        evaluation['risk_adjusted'] = self.calculate_risk_adjusted_metrics(returns)
        
        # 回撤分析
        evaluation['drawdown_analysis'] = self.analyze_drawdowns(returns)
        
        # 分布特征
        evaluation['distribution_stats'] = self.analyze_distribution(returns)
        
        # 时间序列分析
        evaluation['time_series_stats'] = self.analyze_time_series(returns)
        
        # 基准比较
        if benchmark_returns is not None:
            evaluation['benchmark_comparison'] = self.compare_with_benchmark(
                returns, benchmark_returns
            )
        
        # 稳定性分析
        evaluation['stability_analysis'] = self.analyze_stability(returns)
        
        self.evaluation_results = evaluation
        
        return evaluation
    
    def calculate_basic_stats(self, returns: pd.Series) -> Dict[str, float]:
        """计算基本统计指标"""
        
        stats = {
            'total_return': (1 + returns).prod() - 1,
            'annual_return': returns.mean() * 252,
            'annual_volatility': returns.std() * np.sqrt(252),
            'skewness': returns.skew(),
            'kurtosis': returns.kurtosis(),
            'min_return': returns.min(),
            'max_return': returns.max(),
            'positive_ratio': (returns > 0).mean(),
            'negative_ratio': (returns < 0).mean(),
            'zero_ratio': (returns == 0).mean()
        }
        
        return stats
    
    def calculate_risk_adjusted_metrics(self, returns: pd.Series) -> Dict[str, float]:
        """计算风险调整指标"""
        
        annual_return = returns.mean() * 252
        annual_vol = returns.std() * np.sqrt(252)
        
        # 夏普比率
        excess_return = annual_return - self.risk_free_rate
        sharpe_ratio = excess_return / annual_vol if annual_vol > 0 else 0
        
        # 索提诺比率
        downside_returns = returns[returns < 0]
        downside_std = downside_returns.std() * np.sqrt(252) if len(downside_returns) > 0 else 0
        sortino_ratio = excess_return / downside_std if downside_std > 0 else np.inf
        
        # 卡尔马比率
        max_drawdown = self.calculate_max_drawdown(returns)
        calmar_ratio = annual_return / abs(max_drawdown) if max_drawdown != 0 else np.inf
        
        # 信息比率（假设等权基准）
        benchmark_return = returns.mean()  # 简化处理
        active_return = returns - benchmark_return
        tracking_error = active_return.std() * np.sqrt(252)
        information_ratio = (active_return.mean() * 252) / tracking_error if tracking_error > 0 else 0
        
        return {
            'sharpe_ratio': sharpe_ratio,
            'sortino_ratio': sortino_ratio,
            'calmar_ratio': calmar_ratio,
            'information_ratio': information_ratio,
            'excess_return': excess_return,
            'tracking_error': tracking_error
        }
    
    def analyze_drawdowns(self, returns: pd.Series) -> Dict[str, Any]:
        """分析回撤"""
        
        # 计算累计收益
        cumulative = (1 + returns).cumprod()
        
        # 计算运行最大值
        running_max = cumulative.expanding().max()
        
        # 计算回撤
        drawdown = (cumulative - running_max) / running_max
        
        # 最大回撤
        max_drawdown = drawdown.min()
        
        # 回撤持续时间
        drawdown_periods = self.identify_drawdown_periods(drawdown)
        
        # 回撤恢复时间
        recovery_times = self.calculate_recovery_times(drawdown_periods)
        
        return {
            'max_drawdown': max_drawdown,
            'max_drawdown_duration': self.find_max_drawdown_duration(drawdown_periods),
            'avg_drawdown': drawdown[drawdown < 0].mean() if (drawdown < 0).any() else 0,
            'drawdown_volatility': drawdown.std(),
            'drawdown_periods': len(drawdown_periods),
            'avg_recovery_time': np.mean(recovery_times) if recovery_times else 0,
            'max_recovery_time': max(recovery_times) if recovery_times else 0,
            'drawdown_series': drawdown
        }
    
    def identify_drawdown_periods(self, drawdown: pd.Series) -> List[Tuple[int, int]]:
        """识别回撤期"""
        periods = []
        in_drawdown = False
        start_idx = None
        
        for i, dd in enumerate(drawdown):
            if dd < 0 and not in_drawdown:
                start_idx = i
                in_drawdown = True
            elif dd >= 0 and in_drawdown:
                periods.append((start_idx, i))
                in_drawdown = False
        
        # 处理最后一个回撤期
        if in_drawdown:
            periods.append((start_idx, len(drawdown) - 1))
        
        return periods
    
    def calculate_recovery_times(self, drawdown_periods: List[Tuple[int, int]]) -> List[int]:
        """计算恢复时间"""
        recovery_times = []
        
        for start_idx, end_idx in drawdown_periods:
            recovery_time = end_idx - start_idx
            recovery_times.append(recovery_time)
        
        return recovery_times
    
    def find_max_drawdown_duration(self, drawdown_periods: List[Tuple[int, int]]) -> int:
        """找到最长回撤持续时间"""
        if not drawdown_periods:
            return 0
        
        durations = [end - start for start, end in drawdown_periods]
        return max(durations)
    
    def calculate_max_drawdown(self, returns: pd.Series) -> float:
        """计算最大回撤"""
        cumulative = (1 + returns).cumprod()
        running_max = cumulative.expanding().max()
        drawdown = (cumulative - running_max) / running_max
        return drawdown.min()
    
    def analyze_distribution(self, returns: pd.Series) -> Dict[str, Any]:
        """分析收益分布"""
        
        # 正态性检验
        from scipy import stats
        
        # Jarque-Bera检验
        jb_stat, jb_pvalue = stats.jarque_bera(returns.dropna())
        
        # Shapiro-Wilk检验
        if len(returns) <= 5000:  # Shapiro-Wilk检验对样本大小有限制
            sw_stat, sw_pvalue = stats.shapiro(returns.dropna())
        else:
            sw_stat, sw_pvalue = None, None
        
        # 分位数分析
        quantiles = [0.01, 0.05, 0.1, 0.25, 0.5, 0.75, 0.9, 0.95, 0.99]
        quantile_values = returns.quantile(quantiles)
        
        # VaR计算
        var_95 = returns.quantile(0.05)
        var_99 = returns.quantile(0.01)
        
        # CVaR计算
        cvar_95 = returns[returns <= var_95].mean()
        cvar_99 = returns[returns <= var_99].mean()
        
        return {
            'jb_statistic': jb_stat,
            'jb_pvalue': jb_pvalue,
            'sw_statistic': sw_stat,
            'sw_pvalue': sw_pvalue,
            'is_normal': jb_pvalue > 0.05,
            'quantiles': dict(zip(quantiles, quantile_values)),
            'var_95': var_95,
            'var_99': var_99,
            'cvar_95': cvar_95,
            'cvar_99': cvar_99,
            'tail_ratio': abs(quantile_values[0.05]) / abs(quantile_values[0.95]) if quantile_values[0.95] != 0 else np.inf
        }
    
    def analyze_time_series(self, returns: pd.Series) -> Dict[str, Any]:
        """时间序列分析"""
        
        from scipy import stats
        
        # 自相关检验
        from statsmodels.stats.diagnostic import acorr_ljungbox
        
        try:
            # Ljung-Box检验（检验自相关）
            lb_test = acorr_ljungbox(returns.dropna(), lags=10, return_df=True)
            lb_stat = lb_test['lb_stat'].iloc[-1]
            lb_pvalue = lb_test['lb_pvalue'].iloc[-1]
        except:
            lb_stat, lb_pvalue = None, None
        
        # 单位根检验（ADF检验）
        from statsmodels.tsa.stattools import adfuller
        adf_result = adfuller(returns.dropna())
        
        # 计算滚动统计量
        rolling_stats = self.calculate_rolling_stats(returns)
        
        return {
            'lb_statistic': lb_stat,
            'lb_pvalue': lb_pvalue,
            'has_autocorrelation': lb_pvalue < 0.05 if lb_pvalue else None,
            'adf_statistic': adf_result[0],
            'adf_pvalue': adf_result[1],
            'is_stationary': adf_result[1] < 0.05,
            'rolling_stats': rolling_stats
        }
    
    def calculate_rolling_stats(self, returns: pd.Series, window: int = 60) -> Dict[str, pd.Series]:
        """计算滚动统计量"""
        
        rolling_stats = {
            'rolling_mean': returns.rolling(window=window).mean(),
            'rolling_std': returns.rolling(window=window).std(),
            'rolling_sharpe': self.calculate_rolling_sharpe(returns, window),
            'rolling_skewness': returns.rolling(window=window).skew(),
            'rolling_kurtosis': returns.rolling(window=window).kurt()
        }
        
        return rolling_stats
    
    def calculate_rolling_sharpe(self, returns: pd.Series, window: int) -> pd.Series:
        """计算滚动夏普比率"""
        rolling_return = returns.rolling(window=window).mean() * 252
        rolling_vol = returns.rolling(window=window).std() * np.sqrt(252)
        
        excess_return = rolling_return - self.risk_free_rate
        rolling_sharpe = excess_return / rolling_vol
        
        return rolling_sharpe
    
    def compare_with_benchmark(self, strategy_returns: pd.Series, 
                              benchmark_returns: pd.Series) -> Dict[str, float]:
        """与基准比较"""
        
        # 对齐数据
        aligned_data = pd.concat([strategy_returns, benchmark_returns], axis=1).dropna()
        strategy_aligned = aligned_data.iloc[:, 0]
        benchmark_aligned = aligned_data.iloc[:, 1]
        
        # 计算Alpha和Beta
        covariance = np.cov(strategy_aligned, benchmark_aligned)[0, 1]
        benchmark_variance = np.var(benchmark_aligned)
        beta = covariance / benchmark_variance if benchmark_variance > 0 else 0
        
        strategy_annual = strategy_aligned.mean() * 252
        benchmark_annual = benchmark_aligned.mean() * 252
        alpha = strategy_annual - (self.risk_free_rate + beta * (benchmark_annual - self.risk_free_rate))
        
        # 跟踪误差
        tracking_error = (strategy_aligned - benchmark_aligned).std() * np.sqrt(252)
        
        # 信息比率
        active_return = (strategy_aligned - benchmark_aligned).mean() * 252
        information_ratio = active_return / tracking_error if tracking_error > 0 else 0
        
        # 相关性
        correlation = strategy_aligned.corr(benchmark_aligned)
        
        return {
            'alpha': alpha,
            'beta': beta,
            'tracking_error': tracking_error,
            'information_ratio': information_ratio,
            'correlation': correlation,
            'active_return': active_return
        }
    
    def analyze_stability(self, returns: pd.Series) -> Dict[str, Any]:
        """稳定性分析"""
        
        # 分段时间分析
        n_periods = 4
        period_length = len(returns) // n_periods
        
        period_sharpes = []
        period_returns = []
        
        for i in range(n_periods):
            start_idx = i * period_length
            end_idx = min((i + 1) * period_length, len(returns))
            period_returns_series = returns.iloc[start_idx:end_idx]
            
            if len(period_returns_series) > 0:
                period_sharpe = self.calculate_sharpe_ratio(period_returns_series)
                period_return = period_returns_series.mean() * 252
                
                period_sharpes.append(period_sharpe)
                period_returns.append(period_return)
        
        # 计算稳定性指标
        sharpe_stability = np.std(period_sharpes) / abs(np.mean(period_sharpes)) if np.mean(period_sharpes) != 0 else np.inf
        return_stability = np.std(period_returns) / abs(np.mean(period_returns)) if np.mean(period_returns) != 0 else np.inf
        
        # 计算Hurst指数（长期记忆性）
        hurst_exponent = self.calculate_hurst_exponent(returns)
        
        return {
            'period_sharpes': period_sharpes,
            'period_returns': period_returns,
            'sharpe_stability': sharpe_stability,
            'return_stability': return_stability,
            'hurst_exponent': hurst_exponent,
            'has_long_memory': hurst_exponent > 0.5,
            'is_stable': sharpe_stability < 0.5 and return_stability < 0.5
        }
    
    def calculate_hurst_exponent(self, returns: pd.Series, max_lag: int = 100) -> float:
        """计算Hurst指数"""
        
        lags = range(2, min(max_lag, len(returns) // 4))
        tau = [np.sqrt(np.std(np.subtract(returns[lag:], returns[:-lag]))) for lag in lags]
        
        if len(tau) < 2:
            return 0.5
        
        # 线性回归拟合
        poly = np.polyfit(np.log(lags), np.log(tau), 1)
        return poly[0] * 2.0
