def get_technical_data(stock_code, period='D', count=100):
    """
    获取股票的技术指标数据
    """
    # 获取历史数据
    df = get_price(stock_code, count=count, period=period)
    
    # 计算技术指标
    df['MA5'] = df['close'].rolling(5).mean()
    df['MA10'] = df['close'].rolling(10).mean()
    df['MA20'] = df['close'].rolling(20).mean()
    
    # 计算RSI
    df['RSI6'] = talib.RSI(df['close'].values, timeperiod=6)
    df['RSI12'] = talib.RSI(df['close'].values, timeperiod=12)
    
    # 计算MACD
    df['MACD'], df['MACDSignal'], df['MACDHist'] = talib.MACD(
        df['close'].values, fastperiod=12, slowperiod=26, signalperiod=9
    )
    
    # 计算KDJ
    df['K'], df['D'] = talib.STOCH(
        df['high'].values, df['low'].values, df['close'].values,
        fastk_period=9, slowk_period=3, slowd_period=3
    )
    df['J'] = 3 * df['K'] - 2 * df['D']
    
    return df

def calculate_ma_strategy(stock_code, short_period=5, long_period=20):
    """
    移动平均线策略
    """
    # 获取股票数据
    df = get_price(stock_code, count=long_period + 50)
    
    # 计算移动平均线
    df['MA_short'] = df['close'].rolling(short_period).mean()
    df['MA_long'] = df['close'].rolling(long_period).mean()
    
    # 计算均线差值
    df['MA_diff'] = df['MA_short'] - df['MA_long']
    
    # 生成交易信号
    df['signal'] = 0
    df.loc[df['MA_short'] > df['MA_long'], 'signal'] = 1  # 金叉买入
    df.loc[df['MA_short'] < df['MA_long'], 'signal'] = -1  # 死叉卖出
    
    # 计算信号变化
    df['position'] = df['signal'].diff()
    
    return df

def calculate_rsi_strategy(stock_code, period=14, overbought=70, oversold=30):
    """
    RSI相对强弱指标策略
    """
    # 获取股票数据
    df = get_price(stock_code, count=period + 50)
    
    # 计算RSI
    df['RSI'] = talib.RSI(df['close'].values, timeperiod=period)
    
    # 生成交易信号
    df['signal'] = 0
    df.loc[df['RSI'] < oversold, 'signal'] = 1   # 超卖买入
    df.loc[df['RSI'] > overbought, 'signal'] = -1  # 超买卖出
    
    # 避免频繁交易，添加确认条件
    df['RSI_prev'] = df['RSI'].shift(1)
    df.loc[(df['RSI'] < oversold) & (df['RSI_prev'] >= oversold), 'confirmed_signal'] = 1
    df.loc[(df['RSI'] > overbought) & (df['RSI_prev'] <= overbought), 'confirmed_signal'] = -1
    
    return df

def calculate_macd_strategy(stock_code):
    """
    MACD策略
    """
    # 获取股票数据
    df = get_price(stock_code, count=100)
    
    # 计算MACD
    df['MACD'], df['Signal'], df['Histogram'] = talib.MACD(
        df['close'].values, fastperiod=12, slowperiod=26, signalperiod=9
    )
    
    # 生成交易信号
    df['signal'] = 0
    
    # MACD线上穿信号线买入
    df.loc[(df['MACD'] > df['Signal']) & (df['MACD'].shift(1) <= df['Signal'].shift(1)), 'signal'] = 1
    
    # MACD线下穿信号线卖出
    df.loc[(df['MACD'] < df['Signal']) & (df['MACD'].shift(1) >= df['Signal'].shift(1)), 'signal'] = -1
    
    # 零轴过滤
    df.loc[df['MACD'] > 0, 'zero_filter'] = 1  # 多头市场
    df.loc[df['MACD'] <= 0, 'zero_filter'] = 0  # 空头市场
    
    return df

def calculate_kdj_strategy(stock_code, k_period=9, d_period=3, j_period=3):
    """
    KDJ随机指标策略
    """
    # 获取股票数据
    df = get_price(stock_code, count=k_period + 50)
    
    # 计算KDJ
    df['K'], df['D'] = talib.STOCH(
        df['high'].values, df['low'].values, df['close'].values,
        fastk_period=k_period, slowk_period=d_period, slowd_period=d_period
    )
    df['J'] = 3 * df['K'] - 2 * df['D']
    
    # 生成交易信号
    df['signal'] = 0
    
    # K线上穿D线买入（金叉）
    df.loc[(df['K'] > df['D']) & (df['K'].shift(1) <= df['D'].shift(1)) & (df['K'] < 80), 'signal'] = 1
    
    # K线下穿D线卖出（死叉）
    df.loc[(df['K'] < df['D']) & (df['K'].shift(1) >= df['D'].shift(1)) & (df['K'] > 20), 'signal'] = -1
    
    # J值极端情况
    df.loc[df['J'] > 100, 'overbought'] = True  # 严重超买
    df.loc[df['J'] < 0, 'oversold'] = True     # 严重超卖
    
    return df

def multi_indicator_strategy(stock_code, context):
    """
    多指标组合策略
    """
    # 获取股票数据
    df = get_price(stock_code, count=50)
    
    # 计算多个技术指标
    indicators = {}
    
    # 1. 移动平均线
    df['MA5'] = df['close'].rolling(5).mean()
    df['MA10'] = df['close'].rolling(10).mean()
    indicators['MA_signal'] = 1 if df['MA5'].iloc[-1] > df['MA10'].iloc[-1] else 0
    
    # 2. RSI
    df['RSI14'] = talib.RSI(df['close'].values, timeperiod=14)
    rsi_value = df['RSI14'].iloc[-1]
    indicators['RSI_signal'] = 0
    if rsi_value < 30:
        indicators['RSI_signal'] = 1  # 超卖
    elif rsi_value > 70:
        indicators['RSI_signal'] = -1  # 超买
    
    # 3. MACD
    df['MACD'], df['Signal'], _ = talib.MACD(df['close'].values)
    macd_value = df['MACD'].iloc[-1]
    signal_value = df['Signal'].iloc[-1]
    indicators['MACD_signal'] = 1 if macd_value > signal_value else 0
    
    # 4. 成交量
    volume_ma = df['volume'].rolling(5).mean()
    indicators['Volume_signal'] = 1 if df['volume'].iloc[-1] > volume_ma.iloc[-1] else 0
    
    # 综合评分
    total_score = (indicators['MA_signal'] + 
                   indicators['RSI_signal'] + 
                   indicators['MACD_signal'] + 
                   indicators['Volume_signal'])
    
    # 生成最终信号
    if total_score >= 3:  # 强烈买入信号
        final_signal = 1
    elif total_score <= -1:  # 卖出信号
        final_signal = -1
    else:  # 中性
        final_signal = 0
    
    return final_signal, indicators

def adaptive_indicator_strategy(stock_code, context):
    """
    自适应指标策略 - 根据市场状态调整参数
    """
    # 获取更长期的数据判断市场状态
    df_long = get_price(stock_code, count=200)
    df_short = df_long.tail(50)
    
    # 判断市场波动性
    volatility = df_long['close'].pct_change().rolling(20).std().iloc[-1]
    
    # 根据波动性调整参数
    if volatility > 0.03:  # 高波动市场
        ma_short, ma_long = 3, 10
        rsi_period, rsi_overbought, rsi_oversold = 10, 75, 25
    else:  # 低波动市场
        ma_short, ma_long = 10, 30
        rsi_period, rsi_overbought, rsi_oversold = 20, 65, 35
    
    # 计算自适应指标
    df_short[f'MA{ma_short}'] = df_short['close'].rolling(ma_short).mean()
    df_short[f'MA{ma_long}'] = df_short['close'].rolling(ma_long).mean()
    df_short['RSI'] = talib.RSI(df_short['close'].values, timeperiod=rsi_period)
    
    # 生成信号
    current_price = df_short['close'].iloc[-1]
    current_ma_short = df_short[f'MA{ma_short}'].iloc[-1]
    current_ma_long = df_short[f'MA{ma_long}'].iloc[-1]
    current_rsi = df_short['RSI'].iloc[-1]
    
    signal = 0
    
    # 趋势跟踪
    if current_ma_short > current_ma_long and current_price > current_ma_short:
        # RSI不过度超买
        if current_rsi < rsi_overbought:
            signal = 1
    elif current_ma_short < current_ma_long and current_price < current_ma_short:
        # RSI不过度超卖
        if current_rsi > rsi_oversold:
            signal = -1
    
    return signal, {
        'volatility': volatility,
        'ma_params': (ma_short, ma_long),
        'rsi_params': (rsi_period, rsi_overbought, rsi_oversold),
        'current_rsi': current_rsi
    }

def initialize(context):
    """
    初始化函数 - 技术指标策略
    """
    # 设置策略参数
    context.strategy_name = "技术指标组合策略"
    context.benchmark = '000300.XSHG'
    
    # 股票池
    context.stock_pool = ['000001.XSHE', '000002.XSHE', '600000.XSHG']
    
    # 策略参数
    context.params = {
        'lookback_period': 50,      # 回看周期
        'position_size': 0.33,      # 每只股票仓位
        'max_positions': 3,          # 最大持仓数
        'stop_loss': 0.05,          # 止损比例
        'take_profit': 0.15,        # 止盈比例
        'rsi_period': 14,          # RSI周期
        'rsi_overbought': 70,      # RSI超买线
        'rsi_oversold': 30,        # RSI超卖线
        'macd_fast': 12,           # MACD快线
        'macd_slow': 26,           # MACD慢线
        'macd_signal': 9           # MACD信号线
    }
    
    # 状态变量
    context.holding_periods = {}    # 持仓周期
    context.entry_prices = {}       # 入场价格
    context.stop_loss_prices = {}   # 止损价格
    context.take_profit_prices = {} # 止盈价格
    
    print(f"策略初始化完成: {context.strategy_name}")

def handle_data(context, data):
    """
    技术指标策略主函数
    """
    current_date = get_datetime()
    
    for stock in context.stock_pool:
        try:
            # 获取当前价格
            current_price = get_current_data(stock).day_limit
            
            # 获取技术指标信号
            signal, indicators = technical_signal(context, stock)
            
            # 获取当前持仓
            current_position = context.portfolio.positions.get(stock, 0)
            
            # 交易逻辑
            if signal > 0 and current_position == 0:  # 买入信号且无持仓
                # 检查是否满足买入条件
                if check_buy_conditions(context, stock, indicators):
                    order_target_percent(stock, context.params['position_size'])
                    context.entry_prices[stock] = current_price
                    context.holding_periods[stock] = 0
                    
                    # 设置止损止盈
                    context.stop_loss_prices[stock] = current_price * (1 - context.params['stop_loss'])
                    context.take_profit_prices[stock] = current_price * (1 + context.params['take_profit'])
                    
                    print(f"{current_date}: 买入 {stock}, 价格: {current_price:.2f}")
            
            elif signal < 0 and current_position > 0:  # 卖出信号且有持仓
                # 检查是否满足卖出条件
                if check_sell_conditions(context, stock, indicators):
                    order_target(stock, 0)
                    
                    # 清理相关数据
                    if stock in context.entry_prices:
                        del context.entry_prices[stock]
                    if stock in context.holding_periods:
                        del context.holding_periods[stock]
                    if stock in context.stop_loss_prices:
                        del context.stop_loss_prices[stock]
                    if stock in context.take_profit_prices:
                        del context.take_profit_prices[stock]
                    
                    print(f"{current_date}: 卖出 {stock}, 价格: {current_price:.2f}")
            
            elif current_position > 0:  # 有持仓时检查止损止盈
                # 更新持仓周期
                if stock in context.holding_periods:
                    context.holding_periods[stock] += 1
                
                # 检查止损
                if current_price <= context.stop_loss_prices.get(stock, 0):
                    order_target(stock, 0)
                    print(f"{current_date}: 止损卖出 {stock}, 价格: {current_price:.2f}")
                
                # 检查止盈
                elif current_price >= context.take_profit_prices.get(stock, 0):
                    order_target(stock, 0)
                    print(f"{current_date}: 止盈卖出 {stock}, 价格: {current_price:.2f}")
                
                # 检查持仓时间
                elif context.holding_periods.get(stock, 0) > 20:  # 最多持有20天
                    order_target(stock, 0)
                    print(f"{current_date}: 时间止损卖出 {stock}, 价格: {current_price:.2f}")
        
        except Exception as e:
            print(f"处理股票 {stock} 时出错: {str(e)}")
            continue

def technical_signal(context, stock):
    """
    计算技术指标信号
    """
    # 获取历史数据
    df = get_price(stock, count=context.params['lookback_period'])
    
    if len(df) < 30:  # 数据不足
        return 0, {}
    
    signals = {}
    
    # 1. 移动平均线信号
    df['MA5'] = df['close'].rolling(5).mean()
    df['MA10'] = df['close'].rolling(10).mean()
    df['MA20'] = df['close'].rolling(20).mean()
    
    # 多头排列
    if (df['MA5'].iloc[-1] > df['MA10'].iloc[-1] > df['MA20'].iloc[-1] and
        df['MA5'].iloc[-2] <= df['MA10'].iloc[-2]):
        signals['MA'] = 1
    # 空头排列
    elif (df['MA5'].iloc[-1] < df['MA10'].iloc[-1] < df['MA20'].iloc[-1] and
          df['MA5'].iloc[-2] >= df['MA10'].iloc[-2]):
        signals['MA'] = -1
    else:
        signals['MA'] = 0
    
    # 2. RSI信号
    df['RSI'] = talib.RSI(df['close'].values, timeperiod=context.params['rsi_period'])
    current_rsi = df['RSI'].iloc[-1]
    
    if current_rsi < context.params['rsi_oversold']:
        signals['RSI'] = 1
    elif current_rsi > context.params['rsi_overbought']:
        signals['RSI'] = -1
    else:
        signals['RSI'] = 0
    
    # 3. MACD信号
    df['MACD'], df['Signal'], _ = talib.MACD(
        df['close'].values,
        fastperiod=context.params['macd_fast'],
        slowperiod=context.params['macd_slow'],
        signalperiod=context.params['macd_signal']
    )
    
    if (df['MACD'].iloc[-1] > df['Signal'].iloc[-1] and
        df['MACD'].iloc[-2] <= df['Signal'].iloc[-2]):
        signals['MACD'] = 1
    elif (df['MACD'].iloc[-1] < df['Signal'].iloc[-1] and
          df['MACD'].iloc[-2] >= df['Signal'].iloc[-2]):
        signals['MACD'] = -1
    else:
        signals['MACD'] = 0
    
    # 4. 成交量信号
    df['Volume_MA5'] = df['volume'].rolling(5).mean()
    if (df['volume'].iloc[-1] > df['Volume_MA5'].iloc[-1] * 1.5 and
        df['close'].iloc[-1] > df['close'].iloc[-2]):
        signals['Volume'] = 1
    elif (df['volume'].iloc[-1] > df['Volume_MA5'].iloc[-1] * 1.5 and
          df['close'].iloc[-1] < df['close'].iloc[-2]):
        signals['Volume'] = -1
    else:
        signals['Volume'] = 0
    
    # 综合评分
    buy_signals = sum(1 for s in signals.values() if s == 1)
    sell_signals = sum(1 for s in signals.values() if s == -1)
    
    # 最终信号
    if buy_signals >= 3 and sell_signals == 0:  # 至少3个买入信号，无卖出信号
        final_signal = 1
    elif sell_signals >= 3 and buy_signals == 0:  # 至少3个卖出信号，无买入信号
        final_signal = -1
    else:
        final_signal = 0
    
    return final_signal, signals

def optimize_strategy_parameters(stock_code, parameter_ranges):
    """
    策略参数优化
    """
    best_params = None
    best_score = -float('inf')
    results = []
    
    # 参数网格搜索
    for rsi_period in parameter_ranges['rsi_period']:
        for rsi_overbought in parameter_ranges['rsi_overbought']:
            for rsi_oversold in parameter_ranges['rsi_oversold']:
                for stop_loss in parameter_ranges['stop_loss']:
                    for take_profit in parameter_ranges['take_profit']:
                        
                        params = {
                            'rsi_period': rsi_period,
                            'rsi_overbought': rsi_overbought,
                            'rsi_oversold': rsi_oversold,
                            'stop_loss': stop_loss,
                            'take_profit': take_profit
                        }
                        
                        # 回测评估
                        score = backtest_strategy(stock_code, params)
                        results.append((params, score))
                        
                        if score > best_score:
                            best_score = score
                            best_params = params
    
    return best_params, best_score, results

def risk_management(context, stock, position_size):
    """
    风险管理
    """
    # 1. 仓位控制
    max_position_size = min(position_size, 0.3)  # 单只股票最大30%
    
    # 2. 止损设置
    current_price = get_current_data(stock).day_limit
    stop_loss_price = current_price * 0.95  # 5%止损
    
    # 3. 相关性检查
    if len(context.portfolio.positions) > 0:
        # 检查与现有持仓的相关性
        correlation = calculate_correlation(stock, list(context.portfolio.positions.keys()))
        if correlation > 0.8:  # 高度相关，降低仓位
            max_position_size *= 0.5
    
    # 4. 波动性调整
    volatility = calculate_volatility(stock, 20)
    if volatility > 0.05:  # 高波动性，降低仓位
        max_position_size *= (1 - volatility)
    
    return max_position_size, stop_loss_price

def calculate_performance_metrics(returns):
    """
    计算策略绩效指标
    """
    metrics = {}
    
    # 总收益
    metrics['total_return'] = (1 + returns).prod() - 1
    
    # 年化收益
    metrics['annual_return'] = (1 + metrics['total_return']) ** (252 / len(returns)) - 1
    
    # 年化波动率
    metrics['annual_volatility'] = returns.std() * np.sqrt(252)
    
    # 夏普比率
    risk_free_rate = 0.03
    metrics['sharpe_ratio'] = (metrics['annual_return'] - risk_free_rate) / metrics['annual_volatility']
    
    # 最大回撤
    cumulative_returns = (1 + returns).cumprod()
    peak = cumulative_returns.expanding().max()
    drawdown = (cumulative_returns - peak) / peak
    metrics['max_drawdown'] = drawdown.min()
    
    # 胜率
    metrics['win_rate'] = (returns > 0).mean()
    
    # 盈亏比
    winning_returns = returns[returns > 0]
    losing_returns = returns[returns < 0]
    if len(losing_returns) > 0:
        metrics['profit_factor'] = abs(winning_returns.mean() / losing_returns.mean())
    else:
        metrics['profit_factor'] = float('inf')
    
    return metrics
